{
  return (isalpha(c) || isdigit(c) || c == '!' || c == '%' || c == '+' || c ==
	  ',' || c == '-' || c == '.' || c == '/' || c == ':' || c == '@' || c
	  == '^' || c == '_');
}

int isOneDigitSpecialToken(char c)
{
  return (c[0] == ";" || c[0] == "|" || c[0] == "(" || c[0] == ")" || c[0] == 
	"<" || c[0] == ">")
}
